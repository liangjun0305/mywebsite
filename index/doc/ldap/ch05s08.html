8. 安装postfix

groupadd postfix
groupadd postdrop
useradd postfix -g postfix -c "postfix user" -d /nonexistent -s /sbin/nologin
tar zxvf postfix-2.6.5.tar.gz
cd postfix-2.6.5
gzip -d ../postfix-2.6.5-vda-ng-bigquota.patch.gz
patch -p1 < ../postfix-2.6.5-vda-ng-bigquota.patch
make makefiles 'CCARGS=-I/usr/local/include -DHAS_LDAP -DUSE_CYRUS_SASL -DUSE_SASL_AUTH -I/usr/local/include/sasl/ -I/usr/local/BerkeleyDB.4.8/include/ -DUSE_TLS -I/usr/local/ssl/include/' 'AUXLIBS=-L/usr/local/lib -lldap -L/usr/local/lib -llber -L/usr/local/lib/ -lsasl2 -L/usr/local/BerkeleyDB.4.8/lib/ -L/usr/local/ssl/lib -lssl -lcrypto'
make
make install
