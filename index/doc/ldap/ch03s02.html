2. 配置apache

为extmail配置一个虚拟主机

NameVirtualHost 192.168.2.38:80

<VirtualHost 192.168.2.38:80>
        ServerName mail.test.com
        ServerAdmin postmaster@test.com
        DocumentRoot /usr/local/httpd/htdocs/extsuite/
        ScriptAlias /extmail/cgi/ /usr/local/httpd/htdocs/extsuite/extmail/cgi/
        Alias /extmail/ /usr/local/httpd/htdocs/extsuite/extmail/html/
        ScriptAlias /extman/cgi/ /usr/local/httpd/htdocs/extsuite/extman/cgi/
        Alias /extman/ /usr/local/httpd/htdocs/extsuite/extman/html/
        ErrorLog /usr/local/httpd/logs/mail.test.com.error_log
        CustomLog /usr/local/httpd/logs/mail.test.com.access_log common
</VirtualHost>

<Directory /usr/local/httpd/htdocs/extsuite/extmail/cgi>
        Options +ExecCGI
</Directory>

<Directory /usr/local/httpd/htdocs/extsuite/extman/cgi>
        Options +ExecCGI
</Directory>
