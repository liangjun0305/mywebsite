5. 安装apache httpd

tar zxvf httpd-2.2.14.tar.gz
./configure --prefix=/usr/local/httpd --enable-so --with-ssl=/usr/local/ssl/ --with-mpm=worker --enable-rewrite  --enable-mods-shared --enable-track-vars --enable-suexec
make
make install
